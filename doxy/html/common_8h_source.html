142 inline spdlog::level::level_enum from_str(const std::string &name) SPDLOG_NOEXCEPT
143 {
144  int level = 0;
145  for (const auto &level_str : level_string_views)
146  {
147  if (level_str == name)

149  return static_cast<level::level_enum>(level);

151  level++;
